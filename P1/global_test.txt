// Exemple de programa de prova per a la calculadora

# Declaració de variables bàsiques
int a
float b, c
string s
bool l

# Assignacions senzilles
a := 5

c := a + b * 2
s := "Hola"

# Expressions aritmètiques amb precedència
a := 2 + 3 * 4
c := (2 + 3) * 4
b := 2 ^ 3 + 5 % 2

# Expressions booleanes
l := a > b
l := (a <= c) and not (b = 0)

# Concatenació de cadenes
s := s + " món"
s := s + 123   // converteix el número en cadena

# Funcions trigonomètriques
b := sin(PI / 2)
c := cos(0) + tan(PI / 4)

# Funcions de cadenes
s := "Programació"
a := LEN(s)
s := SUBSTR(s; 0; 7)

b := E ^ 2
